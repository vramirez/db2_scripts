restore everything
db2 list db directory| grep alias | awk '{print "db2 restore db "$4" replace history file without prompting"}' | sh
#rollforward
db2 list db directory| grep alias | awk '{print "db2 rollforward db "$4" to end of backup and complete "}'  | sh


#convert db to online
export LOGARCHMETH1=/db2logdir/arch/
db2 list db directory| grep alias | awk -v mylog=$LOGARCHMETH1 '{print "mkdir -p "mylog $4}' | sh
db2 list db directory| grep alias | awk -v mylog=$LOGARCHMETH1 '{print "db2 update db cfg for "$4" using LOGARCHMETH1 disk:"mylog $4}' | sh
db2 list db directory| grep alias | awk '{print "db2 backup db "$4" compress "}' | sh
db2 list db directory| grep alias | awk '{print "db2 backup db "$4" online compress  include logs "}' | sh
